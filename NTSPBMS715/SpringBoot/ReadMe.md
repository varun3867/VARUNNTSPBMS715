            - @Repository
            - @Controller
            - @RestController
            - @ControllerAdvice
            - @Configuration
            - @Bean
            - Annotation for Data Supply
              - @Autowired
              - @Value
              - @Qualifier
              - @Primary
          - java Config Annotations
            - @Named
            - @Inejct
            - @Resource

            - @PostConstruct
            - @PreConstruct
          - Misc Annotations
            - @PropertySource/@PropertySources
            - @ComponentScan
            - @ImportResource
            - @Import
            - @Scope
            - @Lazy
